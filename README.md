$conda create -n 가상환경이름 python=3.10

### 5. 가상환경 실행

$conda activate 가상환경 이름

$conda install pandas numpy matplotlib seaborn scikit-learn ipython jupyter

$jupyter notebook .

### 8. 가상환경 끝내기

$conda deactivate
